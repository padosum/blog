### Algorithms 
```javascript
function solution(rows) {
	let answer = [];
	let prevRow = []; 

	for(let i = 0; i < rows; i++) {
		let row = [];
		for(let j = 0; j < i+1; j++) {
			if(i >= 1) {
				if(j === 0) {
					row.push(prevRow[0])
				} else if(j === prevRow.length) {
					row.push(prevRow[prevRow.length-1])
				} else {
					row.push(prevRow[j-1] + prevRow[j])
				}
			} else {
				row.push(1);
			}
		}
		prevRow = row; 
		answer.push(row);
	}
	return answer;
}  
console.log(solution(5));
```
